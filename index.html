function drawBigRoad() { const canvas = document.getElementById('bigRoad'); const ctx = canvas.getContext('2d'); ctx.clearRect(0, 0, canvas.width, canvas.height); const cell = 48; const visibleCols = Math.floor(canvas.width / cell); let columns = [], currentCol = []; history.forEach(h => { if (h.winner === 'T') { if (currentCol.length > 0) currentCol[currentCol.length-1].tie = (currentCol[currentCol.length-1].tie || 0) + 1; return; } if (currentCol.length > 0 && currentCol[currentCol.length-1].winner === h.winner && currentCol.length < 6) { currentCol.push({winner: h.winner, tie: 0}); } else { if (currentCol.length > 0) columns.push(currentCol); // ✅ 修正點 currentCol = [{winner: h.winner, tie: 0}]; } }); if (currentCol.length > 0) columns.push(currentCol); const startCol = Math.max(0, columns.length - visibleCols); columns.slice(startCol).forEach((col, colIdx) => { col.forEach((bead, rowIdx) => { const x = colIdx * cell + 6; const y = rowIdx * cell + 6; ctx.fillStyle = bead.winner === 'B' ? '#ef4444' : '#3b82f6'; ctx.beginPath(); ctx.arc(x+18, y+18, 18, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = 'white'; ctx.font = 'bold 22px Arial'; ctx.textAlign = 'center'; ctx.fillText(bead.winner === 'B' ? '莊' : '閒', x+18, y+27); if (bead.tie > 0) { ctx.fillStyle = '#22c55e'; ctx.font = 'bold 16px Arial'; ctx.fillText(bead.tie > 1 ? bead.tie : '和', x+18, y+42); } }); }); }
